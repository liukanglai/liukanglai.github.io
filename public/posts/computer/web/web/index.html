function daysToNewYear() {
  const today = new Date();
  const newYear = new Date(today.getFullYear() + 1, 0, 1);
  const diff = newYear - today;
  const days = Math.floor(diff / (1000 * 60 * 60 * 24));
  return days;
}

console.log("距离下一年还有 " + daysToNewYear() + " 天");
